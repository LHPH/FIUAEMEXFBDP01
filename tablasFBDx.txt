CREATE TABLE ACADEMIA(
      NOMBRE_ACADEMIA VARCHAR2(40) CONSTRAINT PK_ACADEMIA PRIMARY KEY,
      NUM_PROF NUMBER(3),
      NUM_MATERIAS NUMBER(3)
);

CREATE TABLE ACTA(
    ID_ACTA NUMBER(3) CONSTRAINT PK_ACTA PRIMARY KEY,
    NOMBRE_ACADEMIA VARCHAR2(40) CONSTRAINT FK_ACTA REFERENCES ACADEMIA(NOMBRE_ACADEMIA),
    ID_REUNION NUMBER(3) UNIQUE,
    TIPO_REUNION VARCHAR2(20),
    LUGAR VARCHAR2(50),
    SEMESTRE VARCHAR2(6),
    HORA_REUNION DATE,
    ASUNTO VARCHAR2(70),
    ACUERDOS VARCHAR2(70),
    ASISTIENTES VARCHAR2(80)
);

CREATE TABLE REUNION(
     ID_REUNION NUMBER(3) CONSTRAINT PK_REUNIONES PRIMARY KEY,
     TIPO_R VARCHAR2(20),
     LUGAR VARCHAR2(30),
     ASUNTO VARCHAR2(100),
     HORA VARCHAR2(10),
     FECHA DATE,
     NOREUNION NUMBER(2),
     SEMESTRE VARCHAR2(6),
     NOMBRE_ACADEMIA CONSTRAINT FK_REUNIONES REFERENCES ACADEMIA(NOMBRE_ACADEMIA)
);

CREATE TABLE PROFESOR(
    INDICE NUMBER(2) CONSTRAINT PK_PROFESOR PRIMARY KEY,
    RFC VARCHAR2(18),
    NOMBRE VARCHAR2(20),
    A_PATERNO VARCHAR2(30),
    A_MATERNO VARCHAR2(30),
    GRADO VARCHAR2(15),
    PUESTO VARCHAR2(15),
    EMAIL VARCHAR2(30)
);

CREATE TABLE ACADEMIA_PROFESOR(
    INDICE NUMBER(2) CONSTRAINT FK_PROFESOR REFERENCES PROFESOR(INDICE),
    NOMBRE_ACADEMIA VARCHAR2(40) CONSTRAINT FK_NOMBREACAD REFERENCES ACADEMIA(NOMBRE_ACADEMIA),
    NOMBREP VARCHAR2(40)
);    

CREATE TABLE UNIDAD_APRENDIZAJE(
    ID_UAPRENDIZAJE NUMBER(2) CONSTRAINT PK_UNIDAD PRIMARY KEY,
    NOMBRE VARCHAR2(50),
    CLAVE_UA VARCHAR2(10),
    T_HORAS VARCHAR2(2),
    ACADEMIA VARCHAR2(40) CONSTRAINT FK_UNIDADES REFERENCES ACADEMIA(NOMBRE_ACADEMIA)
);

CREATE TABLE PROFESOR_UNIDAD(
        INDICE NUMBER(2) CONSTRAINT FK_PROFESOR2 REFERENCES PROFESOR(INDICE),
       ID_UAPRENDIZAJE NUMBER(2) CONSTRAINT FK_UNIDAD REFERENCES UNIDAD_APRENDIZAJE (ID_UAPRENDIZAJE),
       NOMBRE VARCHAR2(50),
      MATERIA VARCHAR2(20),
     ACADEMIA VARCHAR2(40),
     SEMESTRE VARCHAR2(6)
);

CREATE TABLE PROFESOR_ACTA(
     INDICE NUMBER(2) CONSTRAINT FK_PROFESOR3 REFERENCES PROFESOR(INDICE),
     ID_ACTA NUMBER(3) CONSTRAINT FK_ACTA2 REFERENCES ACTA(ID_ACTA),
     NOMBRE VARCHAR2(60),
     SEMESTRE VARCHAR2(5),
     FECHA DATE,
     ACADEMIA VARCHAR2(40),
     CONSTRAINT PK_PROFESOR_ACTA PRIMARY KEY(INDICE,ID_ACTA)
);
